    ```
        using System;
        using System.Net.Http;
        using System.Threading.Tasks;
        public class Csmple0
        {
            public static async Task Input()
            {
                Console.WriteLine("要求する日にちを入力してください:");
                Console.WriteLine("　例)2026年1月21日> 20260121");
                string Day=Console.ReadLine();
                Day="https://hashimagu.sakura.ne.jp/TestPage/Day/"+Day+".json";
                using (HttpClient client = new HttpClient())
                {
                    string json = await client.GetStringAsync(Day);
                    Console.WriteLine(json);
                }
            }
        }
    ```
